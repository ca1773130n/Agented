    - Add after existing composable calls:
      ```ts
      const tourMachine = useTourMachine()
      const settingsRouter = useRouter()

      function handleRestartTour(): void {
        tourMachine.restartTour()
        // Navigate to workspace step's route to immediately show first step
        settingsRouter.push('/settings#general')
        // Trigger the tour to start from idle -> welcome -> workspace
        // restartTour() sends RESTART which goes to idle; then send START + NEXT to reach workspace
        tourMachine.startTour()
        tourMachine.nextStep()  // welcome -> workspace (skip welcome since user is already authenticated)
      }
      ```

    Note: `restartTour()` in useTourMachine.ts calls `localStorage.removeItem(STORAGE_KEY)` then `send({ type: 'RESTART' })` which transitions to idle and clears completedSteps.
